function setRangeType(type) {
  $id('h-range-type').value = type;
  var edEl  = $id('f-end-date');
  var cntEl = $id('f-occ-count');

  if (type === 'end_by') {
    edEl.disabled = false;

    cntEl.disabled = true;
    cntEl.value = '';
  } else {
    cntEl.disabled = false;

    edEl.disabled = true;
    edEl.value = '';
  }
}
